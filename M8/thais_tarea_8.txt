def Even_Numbers(numbers):
    even = []

    for num in numbers:
        if num % 2 == 0:
            even.append(num)

    return even

def Ten_Or_Less(numbers):
    result = []

    for num in numbers:
        if num <= 10:
            result.append(num)

    return result

nums = [3, 15, 8, 22, 10, 1]

resultado = Ten_Or_Less(nums)

print("Números menores o iguales a 10:", resultado)
